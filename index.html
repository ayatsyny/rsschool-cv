void	add_last(t_fdf **head, t_fdf **tail, t_fdf *new_node)
    {

	new_node->prev = *tail;
	new_node->next = NULL;
	if (!*tail)
		*tail = new_node;
	else
	{
		(*tail)->next = new_node;
		*tail = new_node;
	}
	if (!*head)
		*head = *tail;
